function uniteUnique(arr) {
  

    let sortedUnionArr = [];


    for(let i = 0; i < arguments.length; i++) {
      for(let j = 0; j < arguments[i].length; j++) {               
         if(!sortedUnionArr.includes(arguments[i][j])){
          sortedUnionArr.push(arguments[i][j]);
        }
      
      }
    }

  return sortedUnionArr;
}
